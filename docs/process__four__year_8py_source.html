66def get_semester(plan: list[dict], year: str|int, semester: str ) -> list:
67 years = ['First Year', 'Second Year', 'Third Year', 'Fourth Year']
68 if type(year) == str:
69 year = year.upper()

71 raise ValueError("Provided year not 'First Year', 'Second Year', 'Third Year', or 'Fourth Year'")

113def set_contsraints(semester_plan: list, semester_id: int) -> dict:
114
115 semester_dict = {}
116 classes_in_semester = set([])
117 for course_group in semester_plan:
118 # Confirm structure of each course group
119 if len(course_group) != 3:
120 raise SyntaxError("Read file is not in the expected format")
121
122 # Set constraint structure for each course
123 for i, course in enumerate(course_group[0]):
124 semester_dict[course] = {
125 'name': course_group[1][i],
126 'equivalent_courses': set([x for x in course_group[0] if x != course]),
127 'same_semester': set([]),
128 'credits': course_group[2],
129 'semesters': {semester_id}
130 }
131 classes_in_semester.update(course_group[0])
132
133
134 # Set hard constraints as every class in semester not in soft constraints
135 for course in semester_dict.keys():
136 semester_dict[course]['same_semester'] = {x for x in classes_in_semester if x != course and x not in semester_dict[course]['equivalent_courses']}
137 return semester_dict

168def create_constraints(filename: str) -> dict:
169

207def convert_asp(constraints: dict, file: str) -> None:
208 for course in constraints.keys():
209 # Equivalent classes
210 for other in constraints[course]['same_semester']:
211 file.write(f'same_semester({course}, {other}).\n')
212
213 # Same Semester classes
214 for other in constraints[course]['equivalent_courses']:
215 file.write(f'equivalent_courses({course}, {other}).\n')
216
217 # Class in Semester
218 for semester in constraints[course]['semesters']:
219 file.write(f'course_in_semester({semester}, {course}).\n')
220
221 # Predetermined Rules
222 file.write('equivalent_courses(C1, C3) :- equivalent_courses(C1, C2), equivalent_courses(C2, C3).\n')
223 file.write('course_in_semester(S1, C2) :- equivalent_courses(C1, C2), course_in_semester(S1, C1).\n')
224 file.write('same_semester(C1, C3) :- same_semester(C1, C2), same_semester(C2, C3).\n')
225 file.write('same_semester(C1, C3) :- equivalent_courses(C1, C2), same_semester(C2, C3).\n')
