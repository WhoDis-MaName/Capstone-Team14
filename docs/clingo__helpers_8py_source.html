37def convert_time(t):
38 t = int(t)
39 hours = t // 60
40 minutes = t % 60
41 suffix = "AM" if hours < 12 else "PM"
42 hours = hours if hours <= 12 else hours - 12

49def parse_line(line):
50 # Get the scheduled_section/8 predicate - list of all sections and their timeslots
51 scheduled_section_match = re.match(r"scheduled_section\‍((.*?)\‍)", line.strip())
52
53 # Get the schedule_change/4 predicate - list of only the changed sections and their new timeslots
54 schedule_change_match = re.match(r"schedule_change\‍((.*?)\‍)", line.strip())
55
56 if scheduled_section_match:
57 fields = scheduled_section_match.group(1).split(",")
58 if len(fields) != 8:
59 return
60 (
61 subject_course,
62 section,
63 section_id,
64 start,
65 end,
66 days,
67 location,
68 instructor,
69 ) = fields
70 subject = subject_course[:4].upper()
71 course_number = subject_course[4:]
72 section = section[1:]
73 section_id = section_id[1:]
74 time_str = f"{convert_time(start)} - {convert_time(end)}"
75 return (
76 subject,
77 course_number,
78 section,
79 {
80 "Class Number": section_id,
81 "Date": "Aug 26, 2024 - Dec 20, 2024",
82 "Time": time_str,
83 "Days": days.upper(),
84 "Location": location.replace("_", " ").title(),
85 "Instructor": instructor.replace("_", " ").title(),
86 },

90 if len(fields) != 4:
91 return
92 (
93 section_id,

95 end,

97 ) = fields
98 section_id = int(section_id[1:])
99 time_start = datetime.strptime(convert_time(start), "%I:%M%p")
