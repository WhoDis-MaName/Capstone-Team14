26def get_table_rows(table) -> list:
27 rows = []
28 for tr in table.find_all("tr")[0:]:
29 cells = []
30 # grab all td tags in this table row
31 tds = tr.find_all("td")
32 if len(tds) == 0:
33 # if no td tags, search for th tags
34 # can be found especially in wikipedia tables below the table
35 ths = tr.find_all("th")
36 for th in ths:
37 cells.append(th.text.strip())
38 else:
39 # use regular td tags
40 for td in tds:
41 cells.append(td.text.strip())
42 rows.append(cells)
43 return rows
